Save a Google Drive Preview as PDF

Use this when a Google Drive file opens in preview mode and you want to save the visible pages into one PDF.

  1. Open the file preview in your browser.
  2. Open Developer Tools.
  3. Chrome or Firefox: Shift + Ctrl + J on Windows/Linux, or Option + Command + J on Mac.
  4. Microsoft Edge: Shift + Ctrl + I.
  5. Safari: Option + Command + C.
  6. Go to the Console tab.
  7. Paste the script below and run it.
(async () => {
  function wait(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }

  function makeTrustedURL(url) {
    if (window.trustedTypes && trustedTypes.createPolicy) {
      const policy = trustedTypes.createPolicy("pdfPolicy_" + Date.now(), {
        createScriptURL: x => x
      });
      return policy.createScriptURL(url);
    }
    return url;
  }

  async function loadJsPDF() {
    if (window.jspdf?.jsPDF) return window.jspdf.jsPDF;

    await new Promise((resolve, reject) => {
      const s = document.createElement("script");
      s.src = makeTrustedURL("https://unpkg.com/jspdf@2.5.1/dist/jspdf.umd.min.js");
      s.onload = resolve;
      s.onerror = reject;
      document.body.appendChild(s);
    });

    return window.jspdf.jsPDF;
  }

  function getScroller() {
    const candidates = [...document.querySelectorAll("div")].filter(el => {
      const style = getComputedStyle(el);
      return (
        el.scrollHeight > el.clientHeight + 300 &&
        (style.overflowY === "auto" || style.overflowY === "scroll")
      );
    });

    candidates.sort((a, b) => b.clientHeight - a.clientHeight);
    return candidates[0] || document.scrollingElement || document.documentElement;
  }

  function visible(el) {
    const r = el.getBoundingClientRect();
    return r.width > 100 && r.height > 100 && r.bottom > 0 && r.top < window.innerHeight;
  }

  function toDataURLFromImage(img) {
    const w = img.naturalWidth || img.width;
    const h = img.naturalHeight || img.height;
    if (!w || !h) return null;

    const canvas = document.createElement("canvas");
    canvas.width = w;
    canvas.height = h;
    const ctx = canvas.getContext("2d");
    ctx.drawImage(img, 0, 0, w, h);
    return {
      dataUrl: canvas.toDataURL("image/jpeg", 0.95),
      width: w,
      height: h
    };
  }

  function toDataURLFromCanvas(canvasEl) {
    const w = canvasEl.width || canvasEl.clientWidth;
    const h = canvasEl.height || canvasEl.clientHeight;
    if (!w || !h) return null;

    return {
      dataUrl: canvasEl.toDataURL("image/jpeg", 0.95),
      width: w,
      height: h
    };
  }

  function snapshotVisiblePages(scroller, savedPages) {
    const nodes = [
      ...document.querySelectorAll("img, canvas")
    ].filter(el => visible(el));

    let added = 0;

    for (const el of nodes) {
      let key = "";
      let snap = null;

      if (el.tagName === "IMG") {
        if (!/^blob:/.test(el.src)) continue;
        key = "img:" + el.src;
        if (savedPages.has(key)) continue;
        snap = toDataURLFromImage(el);
      } else if (el.tagName === "CANVAS") {
        const rect = el.getBoundingClientRect();
        key = "canvas:" + Math.round(scroller.scrollTop + rect.top) + ":" + el.width + "x" + el.height;
        if (savedPages.has(key)) continue;
        snap = toDataURLFromCanvas(el);
      }

      if (!snap) continue;

      const rect = el.getBoundingClientRect();
      savedPages.set(key, {
        order: scroller.scrollTop + rect.top,
        ...snap
      });
      added++;
    }

    return added;
  }

  const jsPDF = await loadJsPDF();
  const scroller = getScroller();
  const savedPages = new Map();

  console.log("Using scroller:", scroller);

  scroller.scrollTop = 0;
  await wait(1500);

  let previousTop = -1;
  let stuckCount = 0;

  while (true) {
    const added = snapshotVisiblePages(scroller, savedPages);
    console.log("Captured so far:", savedPages.size, "new this step:", added);

    const before = scroller.scrollTop;
    scroller.scrollTop += Math.max(600, Math.floor(scroller.clientHeight * 0.8));
    await wait(1200);

    if (scroller.scrollTop === before || scroller.scrollTop === previousTop) {
      stuckCount++;
    } else {
      stuckCount = 0;
    }

    previousTop = scroller.scrollTop;

    snapshotVisiblePages(scroller, savedPages);

    if (stuckCount >= 3 || scroller.scrollTop + scroller.clientHeight >= scroller.scrollHeight - 10) {
      break;
    }
  }

  scroller.scrollTop = scroller.scrollHeight;
  await wait(1500);
  snapshotVisiblePages(scroller, savedPages);

  const pages = [...savedPages.values()].sort((a, b) => a.order - b.order);

  console.log("Total captured pages:", pages.length);

  if (!pages.length) {
    alert("No pages captured.");
    return;
  }

  const first = pages[0];
  const pdf = new jsPDF({
    orientation: first.width > first.height ? "landscape" : "portrait",
    unit: "pt",
    format: [first.width, first.height]
  });

  for (let i = 0; i < pages.length; i++) {
    const p = pages[i];
    if (i > 0) {
      pdf.addPage([p.width, p.height], p.width > p.height ? "landscape" : "portrait");
    }
    pdf.addImage(p.dataUrl, "JPEG", 0, 0, p.width, p.height);
  }

  pdf.save(`all-pages-${pages.length}.pdf`);
})();
